 *     TreeNode left;

 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }

 *         this.right = right;
 *     }

class Solution {
    int maxDiamer = 0;

    public int diameterOfBinaryTree(TreeNode root) {
        dfs(root);
        return maxDiamer;
    }

    int dfs(TreeNode root){
        if (root == null){
            return 0;
        }

        int leftMax = dfs(root.left);
        int rightMax = dfs(root.right);

        // ååºéå†ä½ç½®é¡ºä¾¿è®¡ç®—æœ€å¤§ç›´å¾„

    } 
}
