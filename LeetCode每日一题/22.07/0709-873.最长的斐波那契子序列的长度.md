class Solution {
    public int lenLongestFibSubseq(int[] arr) {
        // æœ€é•¿çš„æ–æ³¢é‚£å¥‘æ•°åˆ—çš„å­åºåˆ—çš„é•¿åº¦

        HashSet<Integer> set = new HashSet<>();
        for (int i : arr){
            set.add(i);
        }

        int res = 2;
        int tempNum = 0;
        for (int j = 0; j < arr.length - 1; j++){
            for (int i = j + 1; i < arr.length; i++){
                int a = arr[j];
                int b = arr[i];

                int temp = 2;

                while (set.contains(a + b)){
                    tempNum = a;
                    a = b;
                    b = tempNum + b;
                    temp++;
                }
                res = (res > temp)? res : temp;
            }
        }
        return (res > 2)? res : 0;
    }
}
